12     import Model.Aplikasi;
13     import java.awt.event.ActionEvent;
14     import java.awt.event.ActionListener;
15     import Model.Nasabah;
16     import View.Penarikan;
17     import javax.swing.JOptionPane;
18     
19     public class ControllerPenarikan implements ActionListener {
20     
21         private Aplikasi model;
22         private Penarikan view;
23         private Nasabah nasabah;
24     
25         public ControllerPenarikan(Aplikasi model, Nasabah pe) {
26             this.model = model;
27             view = new Penarikan();
28             view.setVisible(true);
29             view.addListener(this);
30     
31             nasabah = pe;
32         }
33     
34         @Override
35         public void actionPerformed(ActionEvent ae) {
36             Object source = ae.getSource();
37             if (source.equals(view.getBtnKembal())) {
38                 new ControllerHome(model, nasabah);
39                 view.dispose();
40             } else if (source.equals(view.getBtnLanjutkan())) {
41                 if (nasabah.getSaldo() - Integer.parseInt(view.getNominal()) > 50000 && Integer.parseInt(view.getNominal()) % 50000 == 0 ) {
42                     int saldo = nasabah.getSaldo() - Integer.parseInt(view.getNominal());
43                     model.updateSaldo(nasabah.getNorek(), saldo);
44                     JOptionPane.showMessageDialog(view, "Penarikan Berhasil");
45                     new ControllerLogin(model);
46                     view.dispose();
47                 }else{
48                     JOptionPane.showMessageDialog(view, "Penarikan Gagal");
49                     new ControllerHome(model,nasabah);
50                     view.dispose();
51                 }
52             }
53         }
54     
55     }
