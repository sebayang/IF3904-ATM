25         public ControllerTransfer(Aplikasi model, Nasabah pe) {
26             this.model = model;

28             view.setVisible(true);
29             view.addListener(this);

31             nasabah = pe;

34         @Override
35         public void actionPerformed(ActionEvent ae) {
36             Object source = ae.getSource();
37             if (source.equals(view.getBtnKembali())) {
38                 new ControllerHome(model, nasabah);
39                 view.dispose();
40             } else if (source.equals(view.getBtnLanjutkan())) {
41                 if (view.getNorekTujuan().equals("") || view.getSaldo().equals("")) {
42                     JOptionPane.showMessageDialog(null, "Semua data harus terisi");
43                 } else {
44                     if (nasabah.getSaldo() - Integer.parseInt(view.getSaldo()) > 50000) {
45                         int norek = Integer.parseInt(view.getNorekTujuan());
46                         if (model.getNasabahByNorek(norek) != null) {
47                             Nasabah n = model.getNasabahByNorek(norek);
48                             int saldo = nasabah.getSaldo() - Integer.parseInt(view.getSaldo());
49                             int saldo2 = n.getSaldo() + Integer.parseInt(view.getSaldo());
50                             System.out.println("a");
51                             model.updateSaldo(nasabah.getNorek(), saldo);
52                             model.updateSaldo(norek, saldo2);
53                             JOptionPane.showMessageDialog(view, "Transfer Berhasil");
54                             new ControllerLogin(model);
55                             view.dispose();
56                         } else {
57                             System.out.println("b");
58                             JOptionPane.showMessageDialog(view, "Nomore Rekening Tujuan Tidak Dapat Ditemukan");
59                         }
60     
61                         view.dispose();
62                     }
63                 }
64     
65             }
66         }
